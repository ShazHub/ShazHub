-- Function to create ESP for a player
local function CreateESP(Player)
    if Player == LocalPlayer or ESPObjects[Player] then return end
    
    local ESPFolder = Instance.new("Folder")
    ESPFolder.Name = Player.Name .. "_ESP"
    ESPFolder.Parent = workspace
    
    local espData = {
        Player = Player,
        Folder = ESPFolder,
        BoxFrame = nil,
        SkeletonParts = {},
        HealthBar = nil,
        UsernameLabel = nil,
    }
    
    -- Create Box
    local BoxFrame = Instance.new("Frame")
    BoxFrame.Name = "Box"
    BoxFrame.BorderColor3 = Color3.fromRGB(0, 255, 0)
    BoxFrame.BorderSizePixel = 2
    BoxFrame.BackgroundTransparency = 1
    BoxFrame.Parent = ScreenGui
    espData.BoxFrame = BoxFrame
    
    -- Create Username Label
    local UsernameLabel = Instance.new("TextLabel")
    UsernameLabel.Name = "Username"
    UsernameLabel.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    UsernameLabel.BackgroundTransparency = 0.5
    UsernameLabel.TextColor3 = Color3.fromRGB(0, 255, 0)
    UsernameLabel.TextSize = 12
    UsernameLabel.Font = Enum.Font.GothamBold
    UsernameLabel.Text = Player.Name
    UsernameLabel.Parent = ScreenGui
    espData.UsernameLabel = UsernameLabel
    
    -- Create Health Bar
    local HealthBarBackground = Instance.new("Frame")
    HealthBarBackground.Name = "HealthBarBG"
    HealthBarBackground.BorderColor3 = Color3.fromRGB(255, 255, 255)
    HealthBarBackground.BorderSizePixel = 1
    HealthBarBackground.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    HealthBarBackground.Parent = ScreenGui
    
    local HealthBar = Instance.new("Frame")
    HealthBar.Name = "HealthBar"
    HealthBar.BorderSizePixel = 0
    HealthBar.BackgroundColor3 = Color3.fromRGB(0, 255, 0)
    HealthBar.Parent = HealthBarBackground
    
    espData.HealthBar = {Background = HealthBarBackground, Bar = HealthBar}
    
    -- Create Skeleton
    local skeletonBones = {
        {"Head", "Torso"},
        {"Torso", "LeftUpperArm"},
        {"LeftUpperArm", "LeftLowerArm"},
        {"LeftLowerArm", "LeftHand"},
        {"Torso", "RightUpperArm"},
        {"RightUpperArm", "RightLowerArm"},
        {"RightLowerArm", "RightHand"},
        {"Torso", "LeftUpperLeg"},
        {"LeftUpperLeg", "LeftLowerLeg"},
        {"LeftLowerLeg", "LeftFoot"},
        {"Torso", "RightUpperLeg"},
        {"RightUpperLeg", "RightLowerLeg"},
        {"RightLowerLeg", "RightFoot"},
    }
    
    for _, bone in pairs(skeletonBones) do
        local line = Instance.new("Frame")
        line.Name = bone[1] .. "_" .. bone[2]
        line.BorderSizePixel = 0
        line.BackgroundColor3 = Color3.fromRGB(0, 255, 0)
        line.Parent = ScreenGui
        table.insert(espData.SkeletonParts, {Line = line, Bone = bone})
    end
    
    ESPObjects[Player] = espData
end

-- Function to remove ESP for a player
local function RemoveESP(Player)
    if not ESPObjects[Player] then return end
    
    local espData = ESPObjects[Player]
    espData.Folder:Destroy()
    espData.BoxFrame:Destroy()
    espData.UsernameLabel:Destroy()
    espData.HealthBar.Background:Destroy()
    
    for _, part in pairs(espData.SkeletonParts) do
        part.Line:Destroy()
    end
    
    ESPObjects[Player] = nil
end

-- Function to update ESP visuals
local function UpdateESP()
    for Player, espData in pairs(ESPObjects) do
        if not Player or not Player.Parent or not Player.Character then
            RemoveESP(Player)
            continue
        end
        
        local Character = Player.Character
        local HumanoidRootPart = Character:FindFirstChild("HumanoidRootPart")
        local Humanoid = Character:FindFirstChild("Humanoid")
        
        if not HumanoidRootPart or not Humanoid then continue end
        
        local Distance = (HumanoidRootPart.Position - LocalPlayer.Character.HumanoidRootPart.Position).Magnitude
        
        if Distance > Config.ESP_Range then
            espData.BoxFrame.Visible = false
            espData.UsernameLabel.Visible = false
            espData.HealthBar.Background.Visible = false
            for _, part in pairs(espData.SkeletonParts) do
                part.Line.Visible = false
            end
            continue
        end
        
        -- Update Box
        local screenPos, onScreen = Camera:WorldToScreenPoint(HumanoidRootPart.Position)
        if onScreen then
            local size = (Camera.Focus.Position - HumanoidRootPart.Position).Magnitude
            local boxSize = math.max(50, 2000 / size)
            
            espData.BoxFrame.Size = UDim2.new(0, boxSize, 0, boxSize * 1.5)
            espData.BoxFrame.Position = UDim2.new(0, screenPos.X - boxSize / 2, 0, screenPos.Y - boxSize * 0.75)
            espData.BoxFrame.Visible = true
            
            -- Update Username
            espData.UsernameLabel.Size = UDim2.new(0, boxSize, 0, 20)
            espData.UsernameLabel.Position = UDim2.new(0, screenPos.X - boxSize / 2, 0, screenPos.Y - boxSize * 0.75 - 20)
            espData.UsernameLabel.Visible = true
            
            -- Update Health Bar
            local healthPercent = Humanoid.Health / Humanoid.MaxHealth
            espData.HealthBar.Background.Size = UDim2.new(0, 8, 0, boxSize * 1.5)
            espData.HealthBar.Background.Position = UDim2.new(0, screenPos.X - boxSize / 2 - 15, 0, screenPos.Y - boxSize * 0.75)
            espData.HealthBar.Bar.Size = UDim2.new(1, 0, healthPercent, 0)
            espData.HealthBar.Bar.BackgroundColor3 = Color3.fromRGB(255 * (1 - healthPercent), 255 * healthPercent, 0)
            espData.HealthBar.Background.Visible = true
            
            -- Update Skeleton
            for _, skeletonPart in pairs(espData.SkeletonParts) do
                local bone = skeletonPart.Bone
                local part1 = Character:FindFirstChild(bone[1])
                local part2 = Character:FindFirstChild(bone[2])
                
                if part1 and part2 then
                    local pos1, onScreen1 = Camera:WorldToScreenPoint(part1.Position)
                    local pos2, onScreen2 = Camera:WorldToScreenPoint(part2.Position)
                    
                    if onScreen1 and onScreen2 then
                        local line = skeletonPart.Line
                        local distance = (Vector2.new(pos1.X, pos1.Y) - Vector2.new(pos2.X, pos2.Y)).Magnitude
                        local angle = math.atan2(pos2.Y - pos1.Y, pos2.X - pos1.X)
                        
                        line.Size = UDim2.new(0, distance, 0, 2)
                        line.Rotation = math.deg(angle)
                        line.Position = UDim2.new(0, pos1.X, 0, pos1.Y)
                        line.Visible = true
                    else
                        skeletonPart.Line.Visible = false
                    end
                else
                    skeletonPart.Line.Visible = false
                end
            end
        else
            espData.BoxFrame.Visible = false
            espData.UsernameLabel.Visible = false
            espData.HealthBar.Background.Visible = false
            for _, part in pairs(espData.SkeletonParts) do
                part.Line.Visible = false
            end
        end
    end
end

-- Function to find closest player
local function GetClosestPlayer()
    local closestPlayer = nil
    local closestDistance = math.huge
    
    for _, Player in pairs(Players:GetPlayers()) do
        if Player == LocalPlayer or not Player.Character then continue end
        
        local Character = Player.Character
        local HumanoidRootPart = Character:FindFirstChild("HumanoidRootPart")
        
        if not HumanoidRootPart then continue end
        
        local Distance = (HumanoidRootPart.Position - LocalPlayer.Character.HumanoidRootPart.Position).Magnitude
        
        if Distance < closestDistance then
            closestDistance = Distance
            closestPlayer = Player
        end
    end
    
    return closestPlayer
end
